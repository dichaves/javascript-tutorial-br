alternar (navegador) {
caso 'Edge':
alerta ("You've the Edge!");

caso 'Chrome':
caso 'Firefox':
caso 'Safari':
caso 'Opera':
alerta ('Ok, nós também apoiamos esses navegadores');

padrão:
alerta ('Esperamos que esta página pareça bem!');
}
`` `
